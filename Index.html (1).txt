<script>
// ===== STATE (in-memory, no localStorage) =====
var pegawai = [];
var absensi = [];


const BULAN=['','Januari','Februari','Maret','April','Mei','Juni','Juli','Agustus','September','Oktober','November','Desember'];


window.onload=function(){
  var today=new Date();
  document.getElementById('abs-tgl').value=fmtDate(today);
  document.getElementById('tgl-hari').textContent=today.toLocaleDateString('id-ID',{weekday:'long',year:'numeric',month:'long',day:'numeric'});
  document.getElementById('rek-bulan').value=today.getMonth()+1;
  document.getElementById('lap-bulan').value=today.getMonth()+1;
  updateDropdown();updateDashboard();renderAbsensiHariIni();renderPegawai();
};


function fmtDate(d){return d.toISOString().split('T')[0]}


function showTab(id,btn){
  document.querySelectorAll('.section').forEach(s=>s.classList.remove('active'));
  document.querySelectorAll('.nav button').forEach(b=>b.classList.remove('active'));
  document.getElementById(id).classList.add('active');
  btn.classList.add('active');
  if(id==='absensi') renderAbsensiHariIni();
  if(id==='pegawai') renderPegawai();
  if(id==='dashboard') updateDashboard();
}


function badge(v){
  var m={Hadir:'badge-h',Izin:'badge-i',Sakit:'badge-s',Alpha:'badge-a'};
  return `<span class="badge ${m[v]||''}">${v}</span>`;
}


function updateDropdown(){
  var sel=document.getElementById('abs-nama'),cur=sel.value;
  sel.innerHTML='<option value="">-- Pilih Pegawai --</option>';
  pegawai.forEach(p=>sel.innerHTML+=`<option value="${p.id}" ${cur==p.id?'selected':''}>${p.nama}</option>`);
}


function updateDashboard(){
  document.getElementById('d-total').textContent=pegawai.length;
  var today=fmtDate(new Date());
  document.getElementById('d-hadir').textContent=absensi.filter(a=>a.tgl===today&&a.apelP==='Hadir').length;
  document.getElementById('d-apel-p').textContent=absensi.filter(a=>a.tgl===today&&a.apelP==='Hadir').length;
  document.getElementById('d-apel-m').textContent=absensi.filter(a=>a.tgl===today&&a.apelM==='Hadir').length;
}


function tambahPegawai(){
  var nip=document.getElementById('pgw-nip').value.trim();
  var nama=document.getElementById('pgw-nama').value.trim();
  var jab=document.getElementById('pgw-jabatan').value.trim();
  var hp=document.getElementById('pgw-hp').value.trim();
  if(!nama){alert('Nama pegawai wajib diisi!');return;}
  pegawai.push({id:Date.now(),nip,nama,jabatan:jab,hp});
  ['pgw-nip','pgw-nama','pgw-jabatan','pgw-hp'].forEach(id=>document.getElementById(id).value='');
  updateDropdown();renderPegawai();updateDashboard();
}


function hapusPegawai(id){
  if(!confirm('Hapus pegawai ini?'))return;
  pegawai=pegawai.filter(p=>p.id!==id);
  updateDropdown();renderPegawai();updateDashboard();
}


function renderPegawai(){
  var w=document.getElementById('tbl-pegawai-wrap');
  if(!pegawai.length){w.innerHTML='<p class="info">Belum ada data pegawai.</p>';return;}
  var h=`<table><tr><th>No</th><th>NIP</th><th>Nama</th><th>Jabatan</th><th>No. HP</th><th>Aksi</th></tr>`;
  pegawai.forEach((p,i)=>h+=`<tr><td>${i+1}</td><td>${p.nip||'-'}</td><td>${p.nama}</td><td>${p.jabatan||'-'}</td><td>${p.hp||'-'}</td><td><button class="btn btn-danger" onclick="hapusPegawai(${p.id})">Hapus</button></td></tr>`);
  w.innerHTML=h+'</table>';
}


function simpanAbsensi(){
  var tgl=document.getElementById('abs-tgl').value;
  var nid=document.getElementById('abs-nama').value;
  var apelP=document.getElementById('abs-apel-p').value;
  var apelM=document.getElementById('abs-apel-m').value;
  var ket=document.getElementById('abs-ket').value;
  if(!tgl||!nid){alert('Tanggal dan nama pegawai wajib diisi!');return;}
  var pgw=pegawai.find(p=>p.id==nid);
  var idx=absensi.findIndex(a=>a.tgl===tgl&&a.pid==nid);
  var rec={pid:nid,nama:pgw.nama,tgl,apelP,apelM,ket};
  if(idx>=0) absensi[idx]=rec; else absensi.push(rec);
  renderAbsensiHariIni();updateDashboard();
  alert('Absensi berhasil disimpan!');
}


function renderAbsensiHariIni(){
  var today=document.getElementById('abs-tgl').value||fmtDate(new Date());
  var data=absensi.filter(a=>a.tgl===today);
  var w=document.getElementById('tbl-absensi-wrap');
  if(!data.length){w.innerHTML='<p class="info">Belum ada absensi untuk tanggal ini.</p>';return;}
  var h=`<table><tr><th>Nama</th><th>Apel Pagi</th><th>Apel Malam</th><th>Keterangan</th></tr>`;
  data.forEach(a=>h+=`<tr><td>${a.nama}</td><td>${badge(a.apelP)}</td><td>${badge(a.apelM)}</td><td>${a.ket||'-'}</td></tr>`);
  w.innerHTML=h+'</table>';
}


function tampilRekap(){
  var bln=parseInt(document.getElementById('rek-bulan').value);
  var thn=parseInt(document.getElementById('rek-tahun').value);
  var data=absensi.filter(a=>{var d=new Date(a.tgl);return d.getMonth()+1===bln&&d.getFullYear()===thn;});
  var rc=document.getElementById('rekap-result');rc.style.display='block';
  document.getElementById('rekap-judul').textContent=`Rekap Kehadiran — ${BULAN[bln]} ${thn}`;
  if(!pegawai.length){document.getElementById('tbl-rekap-wrap').innerHTML='<p class="info">Belum ada data pegawai.</p>';return;}
  var h=`<table><tr><th>No</th><th>Nama</th><th>Hadir</th><th>Izin</th><th>Sakit</th><th>Alpha</th><th>Apel Pagi ✓</th><th>Apel Malam ✓</th></tr>`;
  pegawai.forEach((p,i)=>{
    var pd=data.filter(a=>a.pid==p.id);
    h+=`<tr><td>${i+1}</td><td>${p.nama}</td>
    <td>${pd.filter(a=>a.apelP==='Hadir'||a.apelM==='Hadir').length}</td>
    <td>${pd.filter(a=>a.apelP==='Izin'||a.apelM==='Izin').length}</td>
    <td>${pd.filter(a=>a.apelP==='Sakit'||a.apelM==='Sakit').length}</td>
    <td>${pd.filter(a=>a.apelP==='Alpha'||a.apelM==='Alpha').length}</td>
    <td>${pd.filter(a=>a.apelP==='Hadir').length}</td>
    <td>${pd.filter(a=>a.apelM==='Hadir').length}</td></tr>`;
  });
  document.getElementById('tbl-rekap-wrap').innerHTML=h+'</table>';
}


function getRekapData(bln,thn){
  var data=absensi.filter(a=>{var d=new Date(a.tgl);return d.getMonth()+1===bln&&d.getFullYear()===thn;});
  return pegawai.map((p,i)=>{
    var pd=data.filter(a=>a.pid==p.id);
    return {
      no:i+1,nip:p.nip||'-',nama:p.nama,jabatan:p.jabatan||'-',
      hadir:pd.filter(a=>a.apelP==='Hadir'||a.apelM==='Hadir').length,
      izin:pd.filter(a=>a.apelP==='Izin'||a.apelM==='Izin').length,
      sakit:pd.filter(a=>a.apelP==='Sakit'||a.apelM==='Sakit').length,
      alpha:pd.filter(a=>a.apelP==='Alpha'||a.apelM==='Alpha').length,
      apelP:pd.filter(a=>a.apelP==='Hadir').length,
      apelM:pd.filter(a=>a.apelM==='Hadir').length
    };
  });
}
